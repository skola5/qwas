JavaScript (Web Audio API)

// Example: Simple Echo Effect
const audioCtx = new AudioContext();
const delayNode = audioCtx.createDelay(2);
const feedback = audioCtx.createGain();
feedback.gain.value = 0.5;
delayNode.connect(feedback);
feedback.connect(delayNode);
// Connect to source and destination

Python (Using PyAudio & NumPy)

import pyaudio
import numpy as np

def echo_audio(input_data, delay_samples, feedback):
    output = np.zeros_like(input_data)
    for i in range(len(input_data)):
        output[i] = input_data[i]
        if i >= delay_samples:
            output[i] += feedback * output[i - delay_samples]
    return output

C++ (Simple Delay Line)

#include <vector>

std::vector<float> applyEcho(const std::vector<float>& input, int delay, float feedback) {
    std::vector<float> output = input;
    for (size_t i = delay; i < input.size(); ++i) {
        output[i] += feedback * output[i - delay];
    }
    return output;
}
